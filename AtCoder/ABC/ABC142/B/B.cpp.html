28 template<class T> string to_string(T v) {
29     bool f = true;
30     string res = "{";
31     for (auto &x : v) {
32         if (!f) {
33             res += ", ";
34         }
35         f = false;
36         res += to_string(x);
37     }
38     res += "}";
39     return res;
40 }
